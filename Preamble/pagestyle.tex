% --- Page Geometry ---
\usepackage[margin=1in]{geometry}

% --- Paragraph Setup ---
\onehalfspacing
\setlength{\parindent}{0.5in}
\setlength{\parskip}{0pt}

% --- Graphics Path ---
\graphicspath{{assets/}}

% --- Header and Footer ---
\pagestyle{fancy}
\fancyhf{}
\fancyhead[R]{%
	\ifnum\value{page}>0
	\includegraphics[height=1.5cm]{header.png}
	\fi
}
\fancyhead[L]{\leftmark}
\fancyfoot[C]{%
	\parbox{\linewidth}{%
		\centering
		\thepage\\[5pt]
		\ifnum\value{page}>3
		\makebox[0pt][c]{\includegraphics[width=\paperwidth]{footer.png}}
		\fi
	}%
}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}
\setlength{\headheight}{48pt}

% Apply fancy style to chapter pages as well
\fancypagestyle{plain}{%
	\fancyhead[R]{%
		\ifnum\value{page}>0
		\includegraphics[height=1.5cm]{header.png}
		\fi
	}
	\fancyfoot[C]{%
		\parbox{\linewidth}{%
			\centering
			\thepage\\[5pt]
			\ifnum\value{page}>3
			\makebox[0pt][c]{\includegraphics[width=\paperwidth]{footer.png}}
			\fi
		}%
	}
	\renewcommand{\footrulewidth}{0pt} 
}
