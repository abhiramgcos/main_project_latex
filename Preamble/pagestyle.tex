% --- Page Geometry ---
\usepackage[
    top=1in,
    bottom=1in,
    left=1in,
    right=1in,
    headheight=40pt, % Adjusted from 15pt to accommodate the header.png image
    headsep=0pt    % Space between header and content
]{geometry}

% --- Paragraph Setup ---
\onehalfspacing
\setlength{\parindent}{0.5in}
\setlength{\parskip}{0pt}

% --- Graphics Path ---
\graphicspath{{assets/}}

% --- Header and Footer ---
\pagestyle{fancy}
\fancyhf{}
\fancyhead[R]{%
	\ifnum\value{page}>0
	\includegraphics[height=1.2cm]{header.png}\hspace*{-1.5cm}
	\fi
}
% \fancyhead[L]{\leftmark} % Removed to avoid title repetition on top left
\fancyfoot[C]{%
	\parbox{\linewidth}{%
		\centering
		\thepage\\[5pt]
		\ifnum\value{page}>0
		\makebox[0pt][c]{\includegraphics[width=\paperwidth]{footer.png}}
		\fi
	}%
}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}

% Apply fancy style to chapter pages as well
\fancypagestyle{plain}{%
	\fancyhead[R]{%
		\ifnum\value{page}>0
		\includegraphics[height=1.2cm]{header.png}\hspace*{-1.5cm}
		\fi
	}
	\fancyfoot[C]{%
		\parbox{\linewidth}{%
			\centering
			\thepage\\[5pt]
			\ifnum\value{page}>0
			\makebox[0pt][c]{\includegraphics[width=\paperwidth]{footer.png}}
			\fi
		}%
	}
	\renewcommand{\footrulewidth}{0pt} 
}

% New pagestyle for pages with background logo and no footer
\fancypagestyle{bglogo}{%
	\fancyhead[R]{%
		\ifnum\value{page}>0
		\includegraphics[height=1.2cm]{header.png}\hspace*{-1.5cm}
		\fi
	}
	\fancyfoot[C]{%
		\parbox{\linewidth}{%
			\centering
			\thepage\\[5pt]
		}%
	}
	\renewcommand{\footrulewidth}{0pt}
	\renewcommand{\headrulewidth}{0pt}
}
